```js
// One-off script to generate a 32x32 pixel-art gear icon (Windows 95
// Control Panel style: flat grey gear, black outline, no gradients/AA)
// as a valid PNG, using only Node's built-in zlib (no image libraries).
// Run with: node scripts/generate-settings-icon.js
const fs = require('fs');
const path = require('path');
const zlib = require('zlib');

const SIZE = 32;
const CENTER = (SIZE - 1) / 2; // 15.5
const OUTER_R = 13; // tooth tip radius
const BODY_R = 10.5; // gear body radius (between teeth)
const HOLE_R = 4; // center hole radius
const TEETH = 8;

const BLACK = [0, 0, 0, 255];
const DARK_GRAY = [0x86, 0x89, 0x8d, 255]; // Colors.darkGray
const LIGHT_GRAY = [0xc3, 0xc6, 0xca, 255]; // Colors.lightGray
const TRANSPARENT = [0, 0, 0, 0];

function isToothAngle(angle) {
    // angle in [0, 2*PI). A tooth occupies the middle 45% of each of the
    // 8 wedges around the circle.
    const step = (Math.PI * 2) / TEETH;
    const a = ((angle % step) + step) % step;
    const half = step * 0.225; // 45% of the wedge, centered
    const mid = step / 2;
    return a > mid - half && a < mid + half;
}

function pixelColor(x, y) {
    const dx = x - CENTER;
    const dy = y - CENTER;
    const dist = Math.sqrt(dx * dx + dy * dy);
    const angle = Math.atan2(dy, dx);

    const edge = isToothAngle(angle) ? OUTER_R : BODY_R;

    if (dist > edge) return TRANSPARENT;
    if (dist > edge - 1.1) return BLACK; // 1px outline band
    if (dist < HOLE_R) return TRANSPARENT; // center hole
    if (dist < HOLE_R + 1.1) return BLACK; // hole outline
    if (dist < HOLE_R + 2.5) return LIGHT_GRAY; // inner highlight ring
    return DARK_GRAY; // gear body fill
}

function buildRawRGBA() {
    const raw = Buffer.alloc(SIZE * (1 + SIZE * 4)); // +1 filter byte per row
    let offset = 0;
    for (let y = 0; y < SIZE; y++) {
        raw[offset++] = 0; // filter type 0 (none)
        for (let x = 0; x < SIZE; x++) {
            const [r, g, b, a] = pixelColor(x, y);
            raw[offset++] = r;
            raw[offset++] = g;
            raw[offset++] = b;
            raw[offset++] = a;
        }
    }
    return raw;
}

function crc32(buf) {
    let c;
    const table = crc32.table || (crc32.table = (() => {
        const t = new Uint32Array(256);
        for (let n = 0; n < 256; n++) {
            c = n;
            for (let k = 0; k < 8; k++) {
                c = c & 1 ? 0xedb88320 ^ (c >>> 1) : c >>> 1;
            }
            t[n] = c >>> 0;
        }
        return t;
    })());
    let crc = 0xffffffff;
    for (let i = 0; i < buf.length; i++) {
        crc = table[(crc ^ buf[i]) & 0xff] ^ (crc >>> 8);
    }
    return (crc ^ 0xffffffff) >>> 0;
}

function chunk(type, data) {
    const typeBuf = Buffer.from(type, 'ascii');
    const lenBuf = Buffer.alloc(4);
    lenBuf.writeUInt32BE(data.length, 0);
    const crcBuf = Buffer.alloc(4);
    crcBuf.writeUInt32BE(crc32(Buffer.concat([typeBuf, data])), 0);
    return Buffer.concat([lenBuf, typeBuf, data, crcBuf]);
}

function buildPNG() {
    const signature = Buffer.from([137, 80, 78, 71, 13, 10, 26, 10]);

    const ihdrData = Buffer.alloc(13);
    ihdrData.writeUInt32BE(SIZE, 0); // width
    ihdrData.writeUInt32BE(SIZE, 4); // height
    ihdrData[8] = 8; // bit depth
    ihdrData[9] = 6; // color type: RGBA
    ihdrData[10] = 0; // compression
    ihdrData[11] = 0; // filter
    ihdrData[12] = 0; // interlace
    const ihdr = chunk('IHDR', ihdrData);

    const raw = buildRawRGBA();
    const idatData = zlib.deflateSync(raw);
    const idat = chunk('IDAT', idatData);

    const iend = chunk('IEND', Buffer.alloc(0));

    return Buffer.concat([signature, ihdr, idat, iend]);
}

const outPath = path.join(
    __dirname,
    '..',
    'src',
    'assets',
    'icons',
    'settingsIcon.png'
);
fs.writeFileSync(outPath, buildPNG());
console.log('Wrote', outPath);
```
